-**Question 6**:applying a function

-**Correct Solution**:



typedef int (*_ii2i_fp)(int ,int );

int * apply (int len , int * l1 , int* l2 , _ii2i_fp def,int*ans )

{

    for (int i =0 ; i< len ; i++)

    {

        ans [i]= def (l1[i] , l2[i]);

    }

    return ans ; 

}
